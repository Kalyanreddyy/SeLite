function ensureUnderWebRoot forceReload
storeEval FUDforum.webRoot() webRoot
if LOG.warn( 'Location: ' +window.location.href ) This evaluates to chrome://selenium-ide/content/selenium-ide.xul
endIf But when in getEval, it evaluates to correct URL.
if LOG.warn( "!window.location.href.startsWith( '${webRoot}' ): " +typeof(!window.location.href.startsWith( '${webRoot}' ))+ ': ' +!window.location.href.startsWith( '${webRoot}' ) )
endIf
if LOG.warn( 'forceReload: ' +$forceReload )
endIf
if !window.location.href.startsWith( '${webRoot}' ) || $forceReload
open ${webRoot}
endIf
endFunction
function logOut
call ensureUnderWebRoot
storeElementPresent //ul/li/a[ contains(@href, 'logout=1' ) ] userLoggedIn
if $userLoggedIn
clickAndWait //ul/li/a[ contains(@href, 'logout=1' ) ]
endIf
endFunction
function logInQuick userNameOrAmpersandRole
call ensureUnderWebRoot forceReload=true
call logOut
storeElementPresent //form[ @id='quick_login_form'  ]//input[ @name='quick_login' ] loginFormPresent
if $loginFormPresent
storeEval FUDforum.roleToUser( '${userNameOrAmpersandRole}' ) userName
type //form[ @id='quick_login_form'  ]//input[ @name='quick_login' ] ${userName}
storeEval SeLiteMisc.loginManagerPassword( '${userName}' ) password
type //form[ @id='quick_login_form'  ]//input[ @name='quick_password' ] ${password}
clickAndWait //form[ @id='quick_login_form'  ]//input[ @name='quick_login_submit' ]
verifyElementPresent //ul/li/a[ contains(@href, 'logout=1' ) ] Verify that we are logged in
endIf
endFunction
