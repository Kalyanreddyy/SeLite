script ensureUnderWebRoot
storeEval Drupal.webRoot() drupalWebRoot
if !window.location.href.startsWith( '${drupalWebRoot}' )
open ${drupalWebRoot}
endIf
endScript
script logOut
storeElementPresent //ul[ @id='toolbar-user' ]/li[ contains(@class, 'logout' ) ] userLoggedIn
if qs{$userLoggedIn}
clickAndWait //li[ contains(@class, 'logout') ]/a
endIf
endScript
script logIn userNameOrRole
storeEval Drupal.roleToUser('${userNameOrRole}') userName
storeElementPresent //ul[ @id='toolbar-user' ]/li[ contains(@class, 'logout' ) ] userLoggedIn
if ${userLoggedIn}
clickAndWait //li[ contains(@class, 'logout') ]/a
endIf
storeElementPresent //form[ @id='user-login-form' ]//input[ @id='edit-name' ] userNamePresent
if userNamePresent
storeEval Drupal.webRoot() drupalWebRoot
open ${drupalWebRoot}
endIf
type //form[ @id='user-login-form' ]//input[ @id='edit-name' ] ${userName}
storeEval SeLiteMisc.loginManagerPassword( '${userName}' ) password
type //form[ @id='user-login-form' ]//input[ @id='edit-pass' ] ${password}
clickAndWait //form[ @id='user-login-form' ]//input[ @id='edit-submit' ]
endScript
script createUser
if false
selectFrame //iframe[contains(@title, 'People dialog')]
endIf
typeRandom id=edit-name obj{maxLength:60, type:'word', store:'newUserName'}
typeRandom id=edit-mail obj{maxLength: 254, type:'email', store:'newUserEmail'}
typeRandom id=edit-pass-pass1 obj{type:'password', store:'newUserPassword'}
type id=edit-pass-pass2 qs{$newUserPassword}
clickAndWait id=edit-submit
drupalInsertUser obj{name: $newUserName, mail: $newUserEmail, pass: $newUserPassword}
call logInAsStoredUser userName=$newUserName
endScript
