function logIn loginOrAmpRole
storeEval SeLiteSettings.roleToUser( '${loginOrAmpRole}' ) login
getEval phpMyFAQ.selectUserByLogin( storedVars.login )
clickAndWait //a[ @href='?action=login' ]
type id=faqusername ${login}
type id=faqpassword `phpMyFAQ.selectedUser.pass`
endFunction
function ensureUnderWebRoot forceReload
selectWindow null
if !window.location.href.startsWith( SeLiteSettings.webRoot() ) || $forceReload
open ` SeLiteSettings.webRoot() `
endIf
endFunction
function logOut
call ensureUnderWebRoot forceReload=true
clickAndWait //a[ contains(@href, '`Serendipity.config('indexFile')`?/admin' ) ]
storeElementPresent //a[ contains(@href, 'serendipity[adminModule]=logout' ) ] userLoggedIn
if $userLoggedIn
clickAndWait //a[ contains(@href, 'serendipity[adminModule]=logout' ) ]
endIf
endFunction
function logInAsStoredUser login
readRecord =`{ formula: Serendipity.formulas.authors, username: $login }` user
call logIn loginOrAmpRole=$login, passwd=$user.password
endFunction
