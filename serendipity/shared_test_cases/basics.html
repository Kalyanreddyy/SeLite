function ensureUnderWebRoot forceReload
storeEval SeLiteSettings.webRoot() webRoot
selectWindow null
if !window.location.href.startsWith( '${webRoot}' ) || $forceReload
open ${webRoot}
endIf
endFunction
function logOut
call ensureUnderWebRoot forceReload=true
clickAndWait //a[ contains(@href, '/index.php?/admin' ) ]
storeElementPresent //ul/li/a[ contains(@href, 'serendipity[adminModule]=logout' ) ] userLoggedIn
if $userLoggedIn
clickAndWait //ul/li/a[ contains(@href, 'serendipity[adminModule]=logout' ) ]
endIf
endFunction
function logIn loginOrAmpRole, passwd
call logOut
getEval LOG.info( 'logIn: loginOrAmpRole=${loginOrAmpRole}, passwd=${passwd}, quick=${quick}' )
storeEval SeLiteSettings.roleToUser( '${loginOrAmpRole}' ) login
getEval Serendipity.selectUsername( storedVars.login )
if $passwd===undefined
storeEval SeLiteMisc.loginManagerPassword( '${login}' ) passwd
endIf
getEval LOG.info( 'logIn: login=${login}, passwd=${passwd}, quick=${quick}' )
waitForTimestampDistinctDownToSeconds logIn 10
assertElementPresent //input[ @name='serendipity[user]' ]
type //input[ @name='serendipity[user]' ] ${login}
type //input[ @name='serendipity[pass]' ] ${passwd}
clickAndWait //form[ @action='serendipity_admin.php'  ]//input[ @type='submit' ]
assertElementPresent //ul/li/a[ contains(@href, 'serendipity[adminModule]=logout' ) ] Verify that we are logged in
endFunction
function logInAsStoredUser login, quick
readRecord object{ formula: FUDforum.formulas.users, login: $login } user
call logIn loginOrAmpRole=$login, passwd=$user.passwd, quick=$quick
endFunction
